Bans a user from the guild

Examples
// Bans a user from the guild using ID
Guild.ban('User ID')
  .then((user) => console.log(`Banned: ${user.id}`))
  .catch(console.error);
// Bans a user from the guild using the user object with days and reason.
Guild.ban(user, { days: 7, reason: 'Test' })
  .then((user) => console.log(`Banned: ${user.id}`))
  .catch(console.error);
